// program to pass array to function and print element,without pointers
#include <stdio.h>
void display(int array[],int n)
{
    for(int i=0;i<n;i++)
    printf("%d\n",array[i]);
}
int main()
{
    int arr[5]={1,2,3,4,5};
    display(arr,5);
    return  0;

}
output:1
       2
       3
       4
       5
