//Program to Search an Element Using Binary Search
#include<stdio.h>
int main()
{
int key,mid;
int a[6] = {10, 20, 30, 40, 50 ,60};
printf("enter an element to search:");
scanf("%d",&key);
int low = 0, high = 5;
while(low <= high)
{
 mid = (low + high) / 2;
if(a[mid] == key)
{ printf("%d found at %d", key,mid+1);
return 0;
}
else if(a[mid] > key)
high = mid - 1;
else
low = mid + 1; }
printf("%d  found at %d",key,mid+1);
return 0; 
    

}
output:
enter an element to search: 40
40 found at 4
